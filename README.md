```txt
PROGRAM            = { STATEMENT }, END OF FILE ;
BLOCK              = { STATEMENT } ;
STATEMENT          = ( λ | ASSIGNMENT | CALL | IF | FOR), "\n" ;
ASSIGNMENT         = IDENTIFIER, "=" | "+=" | "-=" | "*=" | "/=", ASSIGNMENT | BOOLEAN_EXPRESSION ;
CALL               = IDENTIFIER, "(", BOOLEAN_EXPRESSION, { ",", BOOLEAN_EXPRESSION }, ")" ;
IF                 = "IF", BOOLEAN_EXPRESSION, BLOCK ;
FOR                = "FOR", ASSIGNMENT, BOOLEAN_EXPRESSION, ASSIGNMENT, BLOCK ;
BOOLEAN_EXPRESSION = BOOLEAN_TERM, { ("==" | "!=" | ">" | "<" | ">=" | "<="), BOOLEAN_TERM } ;
BOOLEAN_TERM       = BOOLEAN_FACTOR, { "||", BOOLEAN_FACTOR } ;
BOOLEAN_FACTOR     = EXPRESSION, { "&&"", EXPRESSION } ;
EXPRESSION         = TERM, { ("+" | "-"), TERM } ;
TERM               = FACTOR, { ( "*" | "/" ), FACTOR } ;
FACTOR             = ({ "+" | "-" | "!" }, FACTOR) | "(" BOOLEAN_EXPRESSION ")" | IDENTIFIER | CALL ;
NUMBER             = DIGIT, { DIGIT } ;
IDENTIFIER         = LETTER, { LETTER | DIGIT | "_" } ;
DIGIT              = ( 0-9 ) ;
LETTER             = ( a-z | A-Z ) ;
```
